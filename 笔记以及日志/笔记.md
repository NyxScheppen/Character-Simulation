## Day1 service，schema，api层初步
### 示例代码
```python
from fastapi import FastAPI
from pydantic import BaseModel

app = FastAPI()

# 这是一个 Pydantic 请求体
class ChatRequest(BaseModel):
    message: str

# 这是一个简单 GET 接口
@app.get("/")
def root():
    return {"message": "hello"}

# 这是另一个简单 GET 接口
@app.get("/characters")
def get_characters():
    return [
        {"id": 1, "name": "小狐狸"},
        {"id": 2, "name": "小狮子"}
    ]

# 这是一个简单 POST 接口
@app.post("/chat")
# 当前端 POST /chat并传来一个请求体
# FastAPI 会把它解析成 ChatRequest
def chat(payload: ChatRequest):
    return {
        # 用 payload.message 取到消息内容
        "reply": f"你刚刚说的是：{payload.message}"
    }
```
### 拆成多个文件
#### 目录结构
```sh
backend/
└── app/
    ├── api/
    │   └── chat.py
    ├── schemas/
    │   └── chat.py
    ├── services/
    │   └── chat_service.py
    └── main.py
```
#### api/chat.py
```python
from fastapi import APIRouter
from app.schemas.chat import ChatRequest
from app.services.chat_service import generate_reply

router = APIRouter(prefix="/api/v1", tags=["chat"])

@router.get("/characters")
def get_characters():
    return [
        {"id": 1, "name": "小狐狸"},
        {"id": 2, "name": "小狮子"}
    ]

@router.post("/chat")
# 这个函数需要输入参数是ChatRequest的格式
def chat(payload: ChatRequest):
    result = generate_reply(payload)
    return result
```
#### schemas/chat.py
```python
# 导入Pydantic的基础类
from pydantic import BaseModel

# 定义请求体结构
class ChatRequest(BaseModel):
    message: str
```
#### main.py
```python
from fastapi import FastAPI
# 导入 api/chat.py 里的 router
from app.api.chat import router as chat_router

# 创建整个应用
app = FastAPI()

@app.get("/")
def root():
    return {"message": "hello"}

# 把 chat.py 里定义的接口注册到整个 FastAPI 应用里。
app.include_router(chat_router)
```
#### services/chat_service.py
```python
from app.schemas.chat import ChatRequest

def generate_reply(payload: ChatRequest) -> dict:
    user_message = payload.message.strip() # 取出用户消息，去掉前后空格

    if not user_message:
        return {"reply": "你还没有输入内容呀～"}

    return {
        "reply": f"小狐狸我呀收到你的消息啦：{user_message}"
    }
```
